1.穷举法时，需要循坏两个列表长度，时间复杂度大

class Solution:
def maxArea(self, height):
    """
    :type height: List[int]
    :rtype: int
    """
    length=len(height)
    mian=[]
    while length>1:
        for i in range(length-1):

            minhei=min(height[0],height[i+1])
            mianji=minhei*(i+1)
            mian.append(mianji)
            resule=max(mian)

        height.pop(0)
        length=length-1
    return resule
